class _speechState extends State<speech> {

/// **第二個畫面：偵測聲音**
class SoundDetectionScreen extends StatefulWidget {
  final String selectedPhoneme;
  final Function(int) onComplete; // 新增回調函數，用於報告完成狀態

  const SoundDetectionScreen({
    super.key,
    required this.selectedPhoneme,
    required this.onComplete,
  });

  @override
  State<SoundDetectionScreen> createState() => _SoundDetectionScreenState();
}

class _SoundDetectionScreenState extends State<SoundDetectionScreen>
    with SingleTickerProviderStateMixin {
  AudioStreamer? _audioStreamer;
  bool _isListening = false;
  double _currentAmplitude = 0.0;
  int _wordCount = 0;
  bool _hasAddedWord = false;

  // 平台特定的振幅閾值（0.0 到 1.0 之間）
  double _amplitudeThreshold = Platform.isIOS ? 0.15 : 0.2;

  // 波形顯示相關
  final List<double> _waveformData = [];
  static const int maxWaveformPoints = 200; // 最大波形點數

  // 峰值偵測相關
  double _lastAmplitude = 0.0;
  int _consecutiveLowSamples = 0;
  static const int minLowSamplesForReset = 1; // 需要多少個低振幅樣本才重置偵測

  // 動畫控制
  late AnimationController _animationController;
  late Animation<double> _scaleAnimation;

  // 倒數計時
  Timer? _countdownTimer;
  int _remainingTime = 10; // 設定倒數 10 秒

  @override
  void initState() {
    super.initState();
    _initializePlatformSpecificSettings();
    _requestPermissions();

    // 動畫：讓氣泡變大縮小
    _animationController = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 500),
      lowerBound: 1.0,
      upperBound: 2,
    );
    _scaleAnimation =
        Tween<double>(begin: 1.0, end: 1.3).animate(_animationController);
  }

  // 添加平台特定設置初始化
  void _initializePlatformSpecificSettings() {
    if (Platform.isIOS) {
      print('iOS 平台：使用優化的振幅閾值設定');
    } else {
      print('Android 平台：使用標準振幅閾值設定');
    }
  }

  Future<void> _requestPermissions() async {
    var status = await Permission.microphone.status;
    if (!status.isGranted) {
      await Permission.microphone.request();
    }

    // iOS 特殊處理：確保權限狀態穩定
    if (Platform.isIOS && status.isGranted) {
      await Future.delayed(const Duration(milliseconds: 300));
      print('iOS 權限確認完成');
    }
  }

  // 處理音頻數據的回調函數
  void _onAudioData(List<double> buffer) {
    if (!_isListening) return;

    // 計算當前緩衝區的最大振幅
    double maxAmplitude = 0.0;
    for (double sample in buffer) {
      maxAmplitude = math.max(maxAmplitude, sample.abs());
    }

    setState(() {
      _currentAmplitude = maxAmplitude;

      // 更新波形數據
      _waveformData.add(maxAmplitude);
      if (_waveformData.length > maxWaveformPoints) {
        _waveformData.removeAt(0); // 移除最舊的數據點
      }

      // 峰值偵測邏輯：檢測振幅突然增大的事件
      if (maxAmplitude > _amplitudeThreshold) {
        if (!_hasAddedWord && _consecutiveLowSamples >= minLowSamplesForReset) {
          _wordCount++;
          _hasAddedWord = true;
          _consecutiveLowSamples = 0;
          print(
              '${Platform.isIOS ? "iOS" : "Android"} 偵測到音節，當前振幅: ${maxAmplitude.toStringAsFixed(3)}');
        }
        _animationController.forward(); // 氣泡放大
      } else {
        _consecutiveLowSamples++;
        if (_consecutiveLowSamples >= minLowSamplesForReset) {
          _hasAddedWord = false;
        }
        _animationController.reverse(); // 氣泡縮小
      }

      _lastAmplitude = maxAmplitude;
    });
  }

  // 處理音頻錯誤的回調函數
  void _onAudioError(dynamic error) {
    debugPrint('音頻串流錯誤 (${Platform.isIOS ? "iOS" : "Android"}): $error');
    _stopListening();
  }

  void _startListening() {
    if (_isListening) return;

    _resetValues();
    _audioStreamer ??= AudioStreamer();

    try {
      // 開始音頻串流
      _audioStreamer!.start(_onAudioData, _onAudioError,
          sampleRate: Platform.isIOS ? 16000 : 44100);

      // 開始倒數計時
      _startCountdown();
    } catch (e) {
      debugPrint('啟動音頻偵測時發生錯誤 (${Platform.isIOS ? "iOS" : "Android"}): $e');
      _stopListening();
    }

    setState(() => _isListening = true);
    print(
        '開始振幅偵測 - 平台: ${Platform.isIOS ? "iOS" : "Android"}, 閾值: $_amplitudeThreshold');
  }

  void _stopListening() {
    _audioStreamer?.stop();
    _audioStreamer = null;
    _countdownTimer?.cancel();
    setState(() {
      _isListening = false;
      _animationController.reverse();
    });
  }

  void _resetValues() {
    setState(() {
      _wordCount = 0;
      _currentAmplitude = 0.0;
      _hasAddedWord = false;
      _remainingTime = 10;
      _waveformData.clear();
      _consecutiveLowSamples = minLowSamplesForReset;
    });
  }

  // **開始倒數計時**
  void _startCountdown() {
    _countdownTimer?.cancel();
    _countdownTimer = Timer.periodic(const Duration(seconds: 1), (timer) {
      if (_remainingTime > 1) {
        setState(() {
          _remainingTime--;
        });
      } else {
        timer.cancel();
        _finishTest(); // 完成測試
      }
    });
  }

  // 完成測試，返回上一畫面
  void _finishTest() {
    _stopListening();
    widget.onComplete(_wordCount);
    Navigator.pop(context);
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: const Color(0xFFF0F4F8),
      appBar: AppBar(
        title: Text("正在測試：${widget.selectedPhoneme}"),
        backgroundColor: Colors.white,
        elevation: 1,
        shadowColor: Colors.black.withOpacity(0.1),
        centerTitle: true,
        iconTheme: const IconThemeData(color: Colors.black87),
      ),
      body: SafeArea(
        child: SingleChildScrollView(
          // 添加滾動功能
          padding: const EdgeInsets.symmetric(horizontal: 24, vertical: 20),
          child: Column(
            children: [
              _buildTopInfoPanel(),
              const SizedBox(height: 30),
              _buildAnimatedBubble(),
              const SizedBox(height: 30),
              _buildWaveformDisplay(), // 新增波形顯示
              const SizedBox(height: 20),
              _buildThresholdSlider(),
              const SizedBox(height: 20),
              _buildStartStopButton(),
            ],
          ),
        ),
      ),
    );
  }

  // UI 組件：頂部資訊面板
  Widget _buildTopInfoPanel() {
    return Row(
      children: [
        Expanded(child: _buildCountdownTimer()),
        const SizedBox(width: 20),
        Expanded(child: _buildWordCountDisplay()),
      ],
    );
  }

  // UI 組件：發音動畫氣泡
  Widget _buildAnimatedBubble() {
    return Column(
      mainAxisSize: MainAxisSize.min,
      children: [
        AnimatedBuilder(
          animation: _scaleAnimation,
          builder: (context, child) {
            return Transform.scale(
              scale: _scaleAnimation.value,
              child: Container(
                width: 180,
                height: 180,
                decoration: BoxDecoration(
                  gradient: const LinearGradient(
                    colors: [Color(0xFF5A9BEE), Color(0xFF4A90E2)],
                    begin: Alignment.topLeft,
                    end: Alignment.bottomRight,
                  ),
                  shape: BoxShape.circle,
                  boxShadow: [
                    BoxShadow(
                      color: const Color(0xFF4A90E2)
                          .withOpacity(0.4 - (_scaleAnimation.value - 1) / 2),
                      blurRadius: 15 + (_scaleAnimation.value - 1) * 10,
                      spreadRadius: (_scaleAnimation.value - 1) * 2,
                    ),
                  ],
                ),
                child: Center(
                  child: Text(
                    widget.selectedPhoneme,
                    style: const TextStyle(
                      fontSize: 70,
                      fontWeight: FontWeight.bold,
                      color: Colors.white,
                      letterSpacing: 2,
                    ),
                  ),
                ),
              ),
            );
          },
        ),
        const SizedBox(height: 30),
        Text(
          _isListening ? "請對著麥克風大聲發音！" : "點擊「開始測試」以進行錄音",
          style: TextStyle(
            fontSize: 18,
            color: Colors.grey[700],
            fontWeight: FontWeight.w500,
          ),
        ),
      ],
    );
  }

  // UI 組件：波形顯示
  Widget _buildWaveformDisplay() {
    return Container(
      padding: const EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: Colors.white,
        borderRadius: BorderRadius.circular(12),
        boxShadow: [
          BoxShadow(
            color: Colors.black.withOpacity(0.05),
            blurRadius: 10,
            offset: const Offset(0, 4),
          )
        ],
      ),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          Row(
            mainAxisAlignment: MainAxisAlignment.spaceBetween,
            children: [
              Text(
                "即時音量波形 (${Platform.isIOS ? 'iOS' : 'Android'})",
                style: const TextStyle(
                  fontSize: 16,
                  fontWeight: FontWeight.w500,
                  color: Colors.black87,
                ),
              ),
              Text(
                "振幅: ${(_currentAmplitude * 100).toInt()}%",
                style: TextStyle(
                  fontSize: 14,
                  fontWeight: _currentAmplitude > _amplitudeThreshold
                      ? FontWeight.bold
                      : FontWeight.normal,
                  color: _currentAmplitude > _amplitudeThreshold
                      ? Colors.green
                      : Colors.black87,
                ),
              ),
            ],
          ),
          const SizedBox(height: 10),
          Container(
            height: 100,
            width: double.infinity,
            decoration: BoxDecoration(
              color: Colors.grey[50],
              borderRadius: BorderRadius.circular(8),
              border: Border.all(color: Colors.grey[300]!),
            ),
            child: ClipRRect(
              borderRadius: BorderRadius.circular(8),
              child: CustomPaint(
                painter: WaveformPainter(
                  waveformData: _waveformData,
                  threshold: _amplitudeThreshold,

/// 自定義波形繪製器
class WaveformPainter extends CustomPainter {
  final List<double> waveformData;
  final double threshold;
  final double currentAmplitude;

  WaveformPainter({
    required this.waveformData,
    required this.threshold,
    required this.currentAmplitude,
  });

  @override
  void paint(Canvas canvas, Size size) {
    if (waveformData.isEmpty) return;

    final paint = Paint()
      ..color = const Color(0xFF4A90E2)
      ..strokeWidth = 1.5
      ..style = PaintingStyle.stroke;

    final thresholdPaint = Paint()
      ..color = Colors.red.withOpacity(0.7)
      ..strokeWidth = 1.0
      ..style = PaintingStyle.stroke;

    final path = Path();

    // 計算每個點的水平間距
    final double stepX = size.width / math.max(1, waveformData.length - 1);

    // 繪製波形
    for (int i = 0; i < waveformData.length; i++) {
      final x = i * stepX;
      final y = size.height - (waveformData[i] * size.height); // 反轉 Y 軸

      if (i == 0) {
        path.moveTo(x, y);
      } else {
        path.lineTo(x, y);
      }
    }

    canvas.drawPath(path, paint);

    // 繪製閾值線
    final thresholdY = size.height - (threshold * size.height);
    canvas.drawLine(
      Offset(0, thresholdY),
      Offset(size.width, thresholdY),
      thresholdPaint,
    );

    // 如果當前振幅超過閾值，在最右側繪製高亮點
    if (currentAmplitude > threshold && waveformData.isNotEmpty) {
      final highlightPaint = Paint()
        ..color = Colors.green
        ..style = PaintingStyle.fill;

      final lastX = (waveformData.length - 1) * stepX;
      final lastY = size.height - (currentAmplitude * size.height);

      canvas.drawCircle(Offset(lastX, lastY), 3.0, highlightPaint);
    }
  }

  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) {
    return true; // 總是重繪以顯示即時數據
  }
}
